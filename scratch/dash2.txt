import { useEffect, useState, useMemo } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { useNavigate } from 'react-router-dom';
import { useAuthStore } from '@klinflow/core/stores/authStore';
import { useDisputeStore } from '@klinflow/core/stores/disputeStore';
import { useAgentStore } from '@klinflow/core/stores/agentStore';
import { useFulfillmentStore } from '@klinflow/core/stores/fulfillmentStore';
import { useMarketplaceStore } from '@klinflow/core/stores/marketplaceStore';
import { useMarketStore } from '@klinflow/core/stores/marketStore';
import { useThemeStore } from '@klinflow/core/stores/themeStore';
import { getThumbnailUrl } from '@klinflow/core/utils/imageUtils';
import { supabase } from '@klinflow/supabase';
import {
  Wallet, Truck, Star, ChevronRight, Copy, Users, Power, Loader2, RefreshCw,
  TrendingUp, Zap, Plus, X, PackageCheck, Store, Receipt, Scale, Activity,
  ArrowUpRight, Clock, UserPlus, Sun, Moon, AlertTriangle, CheckCircle2,
  MapPin, BarChart3, Gauge, CircleDot, Eye, DollarSign, FileWarning, 
  ShieldAlert, ArrowRight, Sparkles, Bot
} from 'lucide-react';
import { toast } from 'sonner';
import {
  PieChart, Pie, Cell, ResponsiveContainer,
  AreaChart, Area, XAxis, YAxis, CartesianGrid, Tooltip,
  ComposedChart, Bar, Line, LineChart
} from 'recharts';
import { MapContainer, TileLayer, Marker, Popup } from 'react-leaflet';
import L from 'leaflet';
import 'leaflet/dist/leaflet.css';

const driverIcon = new L.Icon({
  iconUrl: 'https://cdnjs.cloudflare.com/ajax/libs/leaflet/1.9.4/images/marker-icon.png',
  iconRetinaUrl: 'https://cdnjs.cloudflare.com/ajax/libs/leaflet/1.9.4/images/marker-icon-2x.png',
  shadowUrl: 'https://cdnjs.cloudflare.com/ajax/libs/leaflet/1.9.4/images/marker-shadow.png',
  iconSize: [25, 41],
  iconAnchor: [12, 41],
  popupAnchor: [1, -34],

const DONUT_COLORS = ['#22c55e', '#3b82f6', '#f59e0b', '#94a3b8', '#ef4444'];

export default function CompanyAdminDashboard() {
  const navigate = useNavigate();
  const { logout, profile, fetchProfile, subscribeToProfileChanges, toggleOnline, depositToWallet } = useAuthStore();

  const [showDepositModal, setShowDepositModal] = useState(false);
  const [depositAmount, setDepositAmount] = useState('');
  const [isDepositing, setIsDepositing] = useState(false);

  const earnings = useAgentStore(s => s.earnings);
  const fetchEarnings = useAgentStore(s => s.fetchEarnings);
  const fleetDrivers = useAgentStore(s => s.fleetDrivers);
  const fetchFleetDrivers = useAgentStore(s => s.fetchFleetDrivers);
  const isLoadingFleet = useAgentStore(s => s.isLoadingFleet);
  const [isToggling, setIsToggling] = useState(false);
  const [pendingDriverRequests, setPendingDriverRequests] = useState(0);
  const { isDarkMode, toggleTheme } = useThemeStore();

  // New data sources
  const { dispatchQueue, fetchDispatchQueue, activeFulfillments, fetchActiveFulfillments } = useFulfillmentStore();
  const { listings, fetchListings } = useMarketplaceStore();
  const { materialPrices, fetchMaterialPrices } = useMarketStore();

  const [rfqCount, setRfqCount] = useState(0);
  const [rfqRevenue, setRfqRevenue] = useState(0);

  useEffect(() => {
    fetchProfile();
    fetchEarnings();
    fetchFleetDrivers();
    fetchListings();

    if (profile?.id) {
      subscribeToProfileChanges(profile.id);
      fetchDispatchQueue(profile.id);
      fetchActiveFulfillments(profile.id, 'company');

      // Fetch driver join request counts
      supabase
        .from('company_join_requests')
        .select('*', { count: 'exact', head: true })
        .eq('company_id', profile.id)
        .eq('status', 'pending')
        .then(({ count }) => { if (count !== null) setPendingDriverRequests(count); });

      // Fetch RFQ count & revenue
      supabase
        .from('rfqs')
        .select('id, budget', { count: 'exact' })
        .eq('status', 'open')
        .then(({ data, count }) => {
          if (count !== null) setRfqCount(count);
          if (data) setRfqRevenue(data.reduce((sum: number, r: any) => sum + (r.budget || 0), 0));
        });
    }
    
    fetchMaterialPrices();
  }, [profile?.id, fetchMaterialPrices]);

  const handleCopyCode = () => {
    if (profile?.fleetInviteCode) {
      navigator.clipboard.writeText(profile.fleetInviteCode);
      toast.success('Code Copied!', { description: 'Share this with your new drivers.' });
    }
  };

  const handleToggle = async () => {
    if (isToggling) return;
    setIsToggling(true);
    try {
      await toggleOnline();
      toast.success(profile?.isOnline ? 'Marketplace Closed' : 'Marketplace Open!', {
        description: profile?.isOnline ? 'Your fleet is now hidden.' : 'Your company is now visible to residents.'
      });
    } catch (err) {
      toast.error('Toggle failed');
    } finally {
      setIsToggling(false);
    }
  };

  const handleDeposit = async () => {
    const amount = parseFloat(depositAmount);
    if (!amount || amount <= 0) {
      toast.error('Invalid Amount', { description: 'Please enter a positive amount to deposit.' });
      return;
    }

    setIsDepositing(true);
    try {
      await depositToWallet(amount);
      toast.success('Deposit Successful! 💸', { description: `KSh ${amount.toLocaleString()} added to your company wallet.` });
      setShowDepositModal(false);
      setDepositAmount('');
      fetchProfile();
    } catch (err: any) {
      toast.error('Deposit Failed', { description: err.message });
    } finally {
      setIsDepositing(false);
    }
  };

  if (!profile) return null;

  const maintenanceDrivers = fleetDrivers.filter((d: any) => (d.location as any)?.status === 'maintenance');
  const onlineDrivers = fleetDrivers.filter((d: any) => d.is_online && (d.location as any)?.status !== 'maintenance');
  const enRouteDrivers = onlineDrivers.filter((d: any) => d.is_en_route || (d.location as any)?.status === 'en_route');
  const idleDrivers = onlineDrivers.filter((d: any) => !d.is_en_route && (d.location as any)?.status !== 'en_route');
  const offlineDrivers = fleetDrivers.filter((d: any) => !d.is_online && (d.location as any)?.status !== 'maintenance');

  const activePickups = activeFulfillments.filter(f => !['delivered', 'completed'].includes(f.status));
  const fleetUtilization = fleetDrivers.length > 0 ? Math.round((onlineDrivers.length / fleetDrivers.length) * 100) : 0;
  
  const fleetStatusData = [
    { name: 'Online', value: idleDrivers.length, color: '#22c55e' },
    { name: 'On Route', value: enRouteDrivers.length, color: '#3b82f6' },
    { name: 'Offline', value: offlineDrivers.length, color: '#94a3b8' },
    { name: 'Maintenance', value: maintenanceDrivers.length, color: '#ef4444' }
  ];

  const totalSimulatedDrivers = fleetDrivers.length;

  // Revenue chart data from weeklyData
  const chartData = (earnings.weeklyData || []).map((d: any) => ({
    name: d.day,
    revenue: d.weight * 40, // approximate KSh per kg
    weight: d.weight,
    pickups: Math.floor(d.weight / 15) + Math.floor(Math.random() * 5),
  }));

  // Highest demand material
  const highestDemand = materialPrices.length > 0
    ? materialPrices.reduce((a, b) => (a.price_per_kg > b.price_per_kg ? a : b))
    : null;

  // Fetch disputes
  const { disputes, fetchDisputes } = useDisputeStore();
  useEffect(() => {
    fetchDisputes();
  }, [fetchDisputes]);

  // Attention center alerts
  const alerts = useMemo(() => {
    const items: { icon: any; title: string; desc: string; action: string; actionLabel: string; color: string }[] = [];
    const openDisputes = disputes.filter((d: any) => d.status === 'open');
    if (openDisputes.length > 0) {
      items.push({ icon: FileWarning, title: `${openDisputes.length} Active Disputes`, desc: 'Require immediate resolution', action: '/admin/disputes', actionLabel: 'Resolve', color: 'text-rose-500' });
    }
    const delayedPickups = activeFulfillments.filter(f => {
      if (!f.created_at) return false;
      const hoursOld = (Date.now() - new Date(f.created_at).getTime()) / 3600000;
      return hoursOld > 3 && !['delivered', 'completed'].includes(f.status);
    });
    if (delayedPickups.length > 0) {
      items.push({ icon: Clock, title: `${delayedPickups.length} Delayed Pickups`, desc: 'Require immediate attention', action: '/pickups', actionLabel: 'View', color: 'text-amber-500' });
    }
    if (pendingDriverRequests > 0) {
      items.push({ icon: UserPlus, title: `${pendingDriverRequests} Driver Pending Approval`, desc: 'Approval required to go online', action: '/admin/driver-requests', actionLabel: 'Review', color: 'text-blue-500' });
    }
    if (rfqCount > 0) {
      items.push({ icon: Receipt, title: `${rfqCount} RFQs Available`, desc:"Don't miss potential revenue", action: '/admin/rfqs', actionLabel: 'View', color: 'text-purple-500' });
    }
    if ((profile.walletBalance || 0) < 1000) {
      items.push({ icon: ShieldAlert, title: 'Marketplace Wallet Low', desc: 'Deposit to continue accepting orders', action: '', actionLabel: 'Deposit', color: 'text-rose-500' });
    }
    return items;
  }, [activeFulfillments, pendingDriverRequests, rfqCount, profile.walletBalance]);

  // Map center (default Nairobi)
  const mapCenter: [number, number] = useMemo(() => {
    const withCoords = onlineDrivers.filter((d: any) => d.location?.latitude && d.location?.longitude);
    if (withCoords.length > 0) {
      const lat = withCoords.reduce((sum: number, d: any) => sum + d.location.latitude, 0) / withCoords.length;
      const lng = withCoords.reduce((sum: number, d: any) => sum + d.location.longitude, 0) / withCoords.length;
      return [lat, lng];
    }
    return [-1.2921, 36.8219]; // Nairobi
  }, [onlineDrivers]);

  // Card wrapper
  const Card = ({ children, className = '', onClick }: { children: React.ReactNode; className?: string; onClick?: () => void }) => (
    <div onClick={onClick} className={`bg-white dark:bg-slate-800/60 border border-slate-200 dark:border-slate-700/50 rounded-2xl p-5 shadow-sm transition-all ${onClick ? 'cursor-pointer hover:border-slate-300 dark:hover:border-slate-600' : ''} ${className}`}>
      {children}
    </div>
  );

  const SectionHeader = ({ icon: Icon, title, action, actionLabel = 'View all' }: { icon: any; title: string; action?: string; actionLabel?: string }) => (
    <div className="flex items-center justify-between mb-4">
      <div className="flex items-center gap-2">
        <Icon className="font-medium w-4 h-4 text-slate-500" />
        <h3 className="text-[10px] font-bold text-slate-500 uppercase tracking-widest">{title}</h3>
      </div>
      {action && (
        <button onClick={() => navigate(action)} className="font-medium text-[10px] text-primary capitalize tracking-widest hover:underline flex items-center gap-1">
          {actionLabel} <ArrowRight className="w-3 h-3" />
        </button>
      )}
    </div>
  );

  return (
    <div className="space-y-5 animate-fade-in pb-10">

      {/* ══════════════════════════════════════════════════════════════════════
          ROW 1 — TODAY'S PERFORMANCE (5 KPI CARDS)
      ══════════════════════════════════════════════════════════════════════ */}
      <div>
        <div className="flex items-center gap-2 mb-3">
          <h3 className="text-[10px] font-bold text-slate-500 uppercase tracking-widest">Today's Performance</h3>
          <div className="w-4 h-4 rounded-full bg-slate-100 dark:bg-slate-800 flex items-center justify-center">
            <span className="font-medium text-[8px] text-slate-500">ⓘ</span>
          </div>
        </div>
        
        {(() => {
          const revTrend = earnings.yesterdayPayout ? (((earnings.todayPayout || 0) - earnings.yesterdayPayout) / earnings.yesterdayPayout * 100) : 0;
          const kgTrend = earnings.yesterdayKg ? (((earnings.todayKg || 0) - earnings.yesterdayKg) / earnings.yesterdayKg * 100) : 0;
          const jobsTrend = earnings.yesterdayJobs ? (((earnings.completedToday || 0) - earnings.yesterdayJobs) / earnings.yesterdayJobs * 100) : 0;
          
          return (
            <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-5 gap-3">
              {[
                {
                  label: 'Revenue Today',
                  value: `KSh ${(earnings.todayPayout || 0).toLocaleString()}`,
                  icon: DollarSign,
                  color: 'text-emerald-500',
                  bgColor: 'bg-emerald-50 dark:bg-emerald-500/10',
                  trend: `${revTrend >= 0 ? '+' : ''}${revTrend.toFixed(1)}%`,
                  trendUp: revTrend >= 0,
                },
                {
                  label: 'Collections',
                  value: `${((earnings.todayKg || 0)).toFixed(1)} Kg`,
                  icon: Scale,
                  color: 'text-blue-500',
                  bgColor: 'bg-blue-50 dark:bg-blue-500/10',
                  trend: `${kgTrend >= 0 ? '+' : ''}${kgTrend.toFixed(1)}%`,
                  trendUp: kgTrend >= 0,
                },
                {
                  label: 'Completed Pickups',
                  value: `${earnings.completedToday || 0}`,
                  icon: CheckCircle2,
                  color: 'text-violet-500',
                  bgColor: 'bg-violet-50 dark:bg-violet-500/10',
                  trend: `${jobsTrend >= 0 ? '+' : ''}${jobsTrend.toFixed(1)}%`,
                  trendUp: jobsTrend >= 0,
                },
                {
                  label: 'Fleet Utilization',
                  value: `${fleetUtilization}%`,
                  icon: Gauge,
                  color: 'text-amber-500',
                  bgColor: 'bg-amber-50 dark:bg-amber-500/10',
                  trend: `${onlineDrivers.length}/${fleetDrivers.length}`,
                  trendUp: fleetUtilization > 50,
                },
                {
                  label: 'Fleet Rating',
                  value: profile.rating ? Number(profile.rating).toFixed(1) : 'New',
                  icon: Star,
                  color: 'text-amber-500',
                  bgColor: 'bg-amber-50 dark:bg-amber-500/10',
                  trend: `${profile.rewardPoints || 0} pts`,
                  trendUp: true,
                },
              ].map((kpi) => (
                <Card key={kpi.label}>
                  <div className="flex items-center gap-2 mb-3">
                    <div className={`w-8 h-8 rounded-xl flex items-center justify-center ${kpi.bgColor}`}>
                      <kpi.icon className={`w-4 h-4 ${kpi.color}`} />
                    </div>
                    <p className="font-bold text-[10px] text-slate-600 uppercase tracking-widest">{kpi.label}</p>
                  </div>
                  <div className="flex items-center justify-between">
                    <h2 className="text-xl font-bold text-slate-900 dark:text-white tracking-tighter">{kpi.value}</h2>
                    {kpi.label === 'Company Wallet' && (
                      <button onClick={(e) => { e.stopPropagation(); setShowDepositModal(true); }} className="px-2 py-1 bg-emerald-500 text-white rounded text-[9px] font-bold shadow-sm flex items-center gap-1 hover:bg-emerald-600 transition-colors">
                        <Plus className="w-3 h-3"/> Deposit
                      </button>
                    )}
                  </div>
                  <div className="mt-2 flex items-center gap-1.5">
                    <TrendingUp className={`w-3 h-3 ${kpi.trendUp ? 'text-emerald-500' : 'text-rose-500'} ${kpi.trendUp ? '' : 'rotate-180'}`} />
                    <span className={`text-[10px] ${kpi.trendUp ? 'text-emerald-500' : 'text-rose-500'}`}>{kpi.trend}</span>
                    <span className="font-bold text-[10px] text-slate-600">vs yesterday</span>
                  </div>
                </Card>
              ))}
            </div>
          );
        })()}
        {/* We removed the static KPI list in favor of the dynamic calculation above */}
      </div>

      {/* ══════════════════════════════════════════════════════════════════════
          ROW 2 — OPERATIONS / FLEET STATUS / MARKETPLACE / WALLET
      ══════════════════════════════════════════════════════════════════════ */}
      <div className="grid grid-cols-1 lg:grid-cols-3 gap-3">

        {/* Operations Overview */}
        <Card className="lg:col-span-3">
          <SectionHeader icon={Activity} title="Operations Overview" action="/admin/dispatch" actionLabel="View all operations" />
          <div className="h-[250px] -mx-2 mt-4">
            <ResponsiveContainer width="100%" height="100%">
              <ComposedChart data={chartData}>
                <defs>
                  <linearGradient id="colorRev" x1="0" y1="0" x2="0" y2="1">
                    <stop offset="5%" stopColor="#10b981" stopOpacity={0.3}/>
                    <stop offset="95%" stopColor="#10b981" stopOpacity={0}/>
                  </linearGradient>
                </defs>
                <CartesianGrid strokeDasharray="3 3" stroke={isDarkMode ? '#1e293b' : '#f1f5f9'} vertical={false} />
                <XAxis dataKey="name" tick={{ fontSize: 10, fill: '#94a3b8' }} axisLine={false} tickLine={false} />
                <YAxis yAxisId="left" tick={{ fontSize: 10, fill: '#94a3b8' }} axisLine={false} tickLine={false} width={40} />
                <YAxis yAxisId="right" orientation="right" tick={{ fontSize: 10, fill: '#94a3b8' }} axisLine={false} tickLine={false} width={30} />
                <Tooltip
                  contentStyle={{ background: isDarkMode ? '#1e293b' : '#fff', borderRadius: '8px', fontSize: '12px', border: 'none', boxShadow: '0 4px 6px -1px rgb(0 0 0 / 0.1)' }}
                />
                <Area yAxisId="left" type="monotone" dataKey="revenue" name="Revenue (KSh)" stroke="#10b981" strokeWidth={2} fill="url(#colorRev)" />
                <Bar yAxisId="left" dataKey="weight" name="Collected (Kg)" fill="#3b82f6" radius={[4,4,0,0]} barSize={20} />
                <Line yAxisId="right" type="monotone" dataKey="pickups" name="Pickups" stroke="#f59e0b" strokeWidth={3} dot={{ r: 4 }} />
              </ComposedChart>
            </ResponsiveContainer>
          </div>
        </Card>

        {/* Fleet Status Donut Chart */}
        <Card>
          <SectionHeader icon={Truck} title="Fleet Status" action="/admin/agents" actionLabel="View fleet" />
          <div className="flex items-center gap-6">
            <div className="w-32 h-32 relative shrink-0">
              <ResponsiveContainer width="100%" height="100%">
                <PieChart>
                  <Pie
                    data={fleetStatusData}
                    cx="50%"
                    cy="50%"
                    innerRadius={50}
                    outerRadius={62}
                    paddingAngle={3}
                    dataKey="value"
                    strokeWidth={0}
                  >
                    {fleetStatusData.map((entry, index) => (
                      <Cell key={index} fill={entry.color} />
                    ))}
                  </Pie>
                </PieChart>
              </ResponsiveContainer>
              <div className="absolute inset-0 flex flex-col items-center justify-center">
                <span className="font-bold text-xl text-slate-900 dark:text-white leading-none">{totalSimulatedDrivers}</span>
                <span className="font-bold text-[10px] text-slate-600 uppercase tracking-widest mt-0.5">Total</span>
              </div>
            </div>
            <div className="flex-1 space-y-2.5">
              {fleetStatusData.map((s) => (
                <div key={s.name} className="flex items-center gap-2">
                  <div className="w-2.5 h-2.5 rounded-full shadow-sm shrink-0" style={{ backgroundColor: s.color }} />
                  <div className="flex items-center gap-1.5 flex-1 min-w-0">
                    <span className="font-semibold text-[11px] text-slate-600 dark:text-slate-500 capitalize truncate">{s.name}</span>
                    <span className="font-bold text-sm text-slate-900 dark:text-white shrink-0">{s.value}</span>
                  </div>
                </div>
              ))}
            </div>
          </div>
        </Card>

        {/* Marketplace Opportunities */}
        <Card className="lg:col-span-1">
          <SectionHeader icon={Store} title="Marketplace Opportunities" action="/admin/marketplace" actionLabel="View marketplace" />
          <div className="space-y-4 mt-2">
            
            <div className="flex items-center justify-between p-3 bg-slate-50 dark:bg-slate-800/40 rounded-xl border border-slate-100 dark:border-slate-700/50">
               <div>
                 <p className="font-bold text-[10px] text-slate-500 uppercase tracking-widest mb-1">Market RFQs</p>
                 <span className="font-bold text-lg text-slate-900 dark:text-white">12</span>
               </div>
               <div className="text-right">
                 <p className="font-bold text-[10px] text-slate-500 uppercase tracking-widest mb-1">Value</p>
                 <span className="font-bold text-sm text-emerald-600 dark:text-emerald-400">KSh 450,000</span>
               </div>
            </div>

            <div className="flex items-center justify-between p-3 bg-blue-50 dark:bg-blue-900/10 rounded-xl border border-blue-100 dark:border-blue-900/30">
               <div>
                 <p className="font-bold text-[10px] text-blue-500 uppercase tracking-widest mb-1">My Posted RFQs</p>
                 <span className="font-bold text-lg text-blue-700 dark:text-blue-400">3</span>
               </div>
               <div className="text-right">
                 <p className="font-bold text-[10px] text-blue-500 uppercase tracking-widest mb-1">Value</p>
                 <span className="font-bold text-sm text-blue-700 dark:text-blue-400">KSh 120,000</span>
               </div>
            </div>

            {highestDemand && (
              <div className="pt-3 border-t border-slate-100 dark:border-slate-700/50">
                <div className="flex items-center justify-between mb-2">
                  <div>
                    <p className="font-bold text-[10px] text-slate-500 uppercase tracking-widest mb-0.5">Highest Demand</p>
                    <p className="font-medium text-sm text-slate-900 dark:text-white">{highestDemand.material_name || highestDemand.material}</p>
                  </div>
                  <span className="font-medium inline-block text-[10px] text-amber-600 dark:text-amber-400 bg-amber-50 dark:bg-amber-500/10 px-2 py-0.5 rounded-full">Trending Up</span>
                </div>
                <div className="h-12 w-full mt-1">
                  <ResponsiveContainer width="100%" height="100%">
                    <LineChart data={[{val: 10}, {val: 15}, {val: 13}, {val: 20}, {val: 25}, {val: 22}, {val: 30}]}>
                      <Line type="monotone" dataKey="val" stroke="#f59e0b" strokeWidth={2} dot={false} />
                    </LineChart>
                  </ResponsiveContainer>
                </div>
              </div>
            )}
          </div>
        </Card>

        
      </div>

      {/* ══════════════════════════════════════════════════════════════════════
          ROW 3 — LIVE FLEET MAP / REVENUE CHART / TOP DRIVERS
      ══════════════════════════════════════════════════════════════════════ */}
      <div className="grid grid-cols-1 lg:grid-cols-3 gap-3">

        {/* Live Fleet Map */}
        <Card className="lg:col-span-1 !p-0 relative overflow-hidden min-h-[400px] flex flex-col justify-between">
          <div className="absolute inset-0 z-0">
            <MapContainer
              center={mapCenter}
              zoom={12}
              style={{ height: '100%', width: '100%' }}
              zoomControl={false}
              attributionControl={false}
            >
              <TileLayer url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png" />
              {onlineDrivers
                .filter((d: any) => d.location?.latitude && d.location?.longitude)
                .map((driver: any) => (
                  <Marker
                    key={driver.id}
                    position={[driver.location.latitude, driver.location.longitude]}
                    icon={driverIcon}
                  >
                    <Popup>
                      <span className="font-medium text-sm">{driver.name}</span>
                      <br />
                      <span className="font-bold text-xs text-slate-600">{driver.location?.estate || 'On duty'}</span>
                    </Popup>
                  </Marker>
                ))}
            </MapContainer>
          </div>

          <div className="relative z-10 p-4 flex items-start justify-between pointer-events-none">
            <div className="flex items-center gap-2 bg-white/90 dark:bg-slate-900/90 backdrop-blur-md px-3 py-2 rounded-xl border border-slate-200/50 dark:border-slate-700/50 shadow-sm pointer-events-auto">
              <MapPin className="font-medium w-4 h-4 text-slate-500" />
              <h3 className="text-[10px] font-bold text-slate-500 uppercase tracking-widest">Live Fleet Map</h3>
            </div>
            <span className="font-medium flex items-center gap-1.5 px-2.5 py-1 bg-emerald-500/90 backdrop-blur-md text-white text-[10px] rounded-full shadow-sm pointer-events-auto">
              <div className="w-1.5 h-1.5 rounded-full bg-white animate-pulse" />
              Live
            </span>
          </div>

          <div className="relative z-10 p-4 mt-auto pointer-events-none">
            <div className="bg-white/90 dark:bg-slate-900/90 backdrop-blur-md rounded-xl border border-slate-200/50 dark:border-slate-700/50 p-3 shadow-lg pointer-events-auto w-[90%] max-w-[300px] mx-auto mb-2">
              <div className="grid grid-cols-3 gap-2 mb-2">
                {[
                  { icon: Truck, label: 'En Route', value: enRouteDrivers.length, color: 'text-blue-500' },
                  { icon: Users, label: 'On Duty', value: onlineDrivers.length, color: 'text-emerald-500' },
                  { icon: CircleDot, label: 'Idle', value: idleDrivers.length, color: 'text-amber-500' },
                ].map((stat) => (
                  <div key={stat.label} className="flex flex-col items-center text-center gap-1">
                    <div className="flex items-center gap-1">
                      <stat.icon className={`w-3 h-3 ${stat.color}`} />
                      <p className="font-medium text-sm text-slate-900 dark:text-white leading-none">{stat.value}</p>
                    </div>
                    <p className="font-medium text-[8px] text-slate-500 dark:text-slate-500 uppercase tracking-widest leading-tight">{stat.label}</p>
                  </div>
                ))}
              </div>
              <button onClick={() => navigate('/admin/agents')} className="font-medium pt-2 border-t border-slate-200/50 dark:border-slate-700/50 w-full text-[10px] text-primary hover:underline flex items-center justify-center gap-1">
                View full map <ArrowRight className="w-3 h-3" />
              </button>
            </div>
          </div>
        </Card>

        {/* Revenue Overview Chart */}
        <Card className="lg:col-span-1">
          <div className="flex items-center justify-between mb-4">
            <div className="flex items-center gap-2">
              <BarChart3 className="font-medium w-4 h-4 text-slate-500" />
              <h3 className="text-[10px] font-bold text-slate-500 uppercase tracking-widest">Revenue Overview</h3>
            </div>
            <span className="font-bold text-[10px] text-slate-500 bg-slate-50 dark:bg-slate-800 px-2 py-1 rounded-lg">Last 30 Days ▾</span>
          </div>

          <div className="mb-2">
            <p className="font-bold text-xs text-slate-600 dark:text-slate-500 mb-1">Total Revenue</p>
            <div className="flex items-baseline gap-2">
              <h2 className="text-xl font-bold text-slate-900 dark:text-white tracking-tighter">KSh {(earnings.total || 0).toLocaleString()}</h2>
              <span className="font-medium text-[10px] text-emerald-500 flex items-center gap-0.5">
                <TrendingUp className="w-3 h-3" /> +18%
              </span>
              <span className="font-bold text-[10px] text-slate-500">vs previous 30 days</span>
            </div>
          </div>

          <div className="h-[250px] -mx-2">
            <ResponsiveContainer width="100%" height="100%">
              <AreaChart data={chartData}>
                <defs>
                  <linearGradient id="colorRevenue" x1="0" y1="0" x2="0" y2="1">
                    <stop offset="5%" stopColor="#22c55e" stopOpacity={0.2} />
                    <stop offset="95%" stopColor="#22c55e" stopOpacity={0} />
                  </linearGradient>
                </defs>
                <CartesianGrid strokeDasharray="3 3" stroke={isDarkMode ? '#1e293b' : '#f1f5f9'} />
                <XAxis dataKey="name" tick={{ fontSize: 10, fill: '#94a3b8' }} axisLine={false} tickLine={false} />
                <YAxis tick={{ fontSize: 10, fill: '#94a3b8' }} axisLine={false} tickLine={false} width={40} tickFormatter={(v) => `${(v / 1000).toFixed(0)}K`} />
                <Tooltip
                  contentStyle={{
                    background: isDarkMode ? '#1e293b' : '#fff',
                    border: isDarkMode ? '1px solid #334155' : '1px solid #e2e8f0',
                    borderRadius: '12px',
                    fontSize: '11px',
                  }}
                  formatter={(value: any) => [`KSh ${value.toLocaleString()}`, 'Revenue']}
                />
                <Area type="monotone" dataKey="revenue" stroke="#22c55e" strokeWidth={2} fill="url(#colorRevenue)" />
              </AreaChart>
            </ResponsiveContainer>
          </div>

          <div className="grid grid-cols-3 gap-3 mt-4 pt-3 border-t border-slate-100 dark:border-slate-700/50">
            <div>
              <p className="font-medium text-[9px] text-slate-500 uppercase tracking-widest mb-1">Collections</p>
              <p className="font-medium text-sm text-slate-900 dark:text-white">{(fleetDrivers.reduce((sum, d) => sum + (d.collected_kg || 0), 0)).toLocaleString()} Kg</p>
            </div>
            <div>
              <p className="font-medium text-[9px] text-slate-500 uppercase tracking-widest mb-1">Payouts</p>
              <p className="font-medium text-sm text-slate-900 dark:text-white">KSh {(fleetDrivers.reduce((sum, d) => sum + (d.payout_amount || 0), 0) / 1000).toFixed(0)}K</p>
            </div>
            <div>
              <p className="font-medium text-[9px] text-slate-500 uppercase tracking-widest mb-1">Transactions</p>
              <p className="font-medium text-sm text-slate-900 dark:text-white">{fleetDrivers.reduce((sum, d) => sum + (d.completed_jobs || 0), 0)}</p>
            </div>
          </div>
          <button onClick={() => navigate('/admin/finance')} className="font-medium mt-3 text-[10px] text-primary hover:underline flex items-center gap-1">
            View full analytics <ArrowRight className="w-3 h-3" />
          </button>
        </Card>

        {/* Top Drivers Leaderboard */}
        <Card className="lg:col-span-1">
          <div className="flex items-center justify-between mb-4">
            <div className="flex items-center gap-2">
              <Users className="font-medium w-4 h-4 text-slate-500" />
              <h3 className="text-[10px] font-bold text-slate-500 uppercase tracking-widest">Top Drivers</h3>
            </div>
            <button onClick={() => navigate('/admin/agents')} className="font-medium text-[10px] text-primary capitalize tracking-widest hover:underline">View all</button>
          </div>

          <div className="space-y-4">
            {fleetDrivers.length > 0 ? fleetDrivers.slice().sort((a: any, b: any) => (b.completed_jobs || 0) - (a.completed_jobs || 0)).slice(0, 3).map((driver: any, idx: number) => {
              const medals = ['🥇', '🥈', '🥉'];
              return (
                <div key={driver.id} className="flex items-center gap-3">
                  <span className="font-medium text-lg">{medals[idx]}</span>
                  <div className="relative shrink-0">
                    <div className="font-medium w-10 h-10 rounded-full flex items-center justify-center text-sm border-2 shadow-sm overflow-hidden bg-emerald-100 text-emerald-700 border-emerald-500">
                      {driver.avatar_url ? (
                        <img
                          src={getThumbnailUrl(driver.avatar_url, { width: 200 })}
                          loading="lazy"
                          alt={driver.name}
                          className="w-full h-full object-cover"
                        />
                      ) : (
                        driver.name?.charAt(0).toUpperCase()
                      )}
                    </div>
                  </div>
                  <div className="flex-1 min-w-0">
                    <p className="font-medium text-sm text-slate-900 dark:text-white truncate">{driver.name}</p>
                    <div className="flex items-center gap-1 mt-0.5">
                      <Star className="font-medium w-3 h-3 text-amber-500 fill-amber-500" />
                      <span className="font-bold text-[10px] text-slate-500">{driver.rating ? Number(driver.rating).toFixed(1) : '—'}</span>
                    </div>
                  </div>
                  <div className="text-right">
                    <p className="font-medium text-sm text-slate-900 dark:text-white">{driver.completed_jobs || 0}</p>
                    <p className="font-medium text-[9px] text-slate-500 uppercase tracking-widest">Pickups</p>
                  </div>
                  <div className="text-right">
                    <p className="font-medium text-sm text-slate-900 dark:text-white">{(driver.collected_kg || 0).toLocaleString()} Kg</p>
                    <p className="font-medium text-[9px] text-slate-500 uppercase tracking-widest">Collected</p>
                  </div>
                </div>
              );
            }) : (
              <div className="flex flex-col items-center justify-center py-8 text-center">
                <Users className="font-medium w-8 h-8 text-slate-300 dark:text-slate-600 mb-2" />
                <p className="font-bold text-xs text-slate-600">No driver data yet</p>
              </div>
            )}
          </div>

          <button onClick={() => navigate('/admin/agents')} className="font-medium mt-4 text-[10px] text-primary hover:underline flex items-center gap-1">
            View driver performance <ArrowRight className="w-3 h-3" />
          </button>
        </Card>
      </div>

      {/* ══════════════════════════════════════════════════════════════════════
          ROW 4 — MATERIAL MARKET INTELLIGENCE / ATTENTION CENTER
      ══════════════════════════════════════════════════════════════════════ */}
      <div className="grid grid-cols-1 lg:grid-cols-2 gap-3">

        {/* Material Market Intelligence */}
        <Card>
          <div className="flex items-center justify-between mb-4">
            <div className="flex items-center gap-2">
              <Scale className="font-medium w-4 h-4 text-slate-500" />
              <h3 className="text-[10px] font-bold text-slate-500 uppercase tracking-widest">Material Market Intelligence</h3>
            </div>
            <button onClick={() => navigate('/admin/market-intelligence')} className="font-medium text-[10px] text-primary hover:underline flex items-center gap-1">
              View all prices <ArrowRight className="w-3 h-3" />
            </button>
          </div>

          <div className="grid grid-cols-2 md:grid-cols-4 gap-3 mb-4">
            {(materialPrices.length > 0 ? materialPrices.slice(0, 8).map(m => ({
              material: m.material_name,
              avgPrice: m.price_per_kg,
              trend: m.change_pct || 0
            })) : [
              { material: 'PET Bottles', avgPrice: 42, trend: 12 },
              { material: 'HDPE', avgPrice: 38, trend: 8 },
              { material: 'Cardboard', avgPrice: 28, trend: 3 },
              { material: 'Aluminum', avgPrice: 120, trend: 5 }
            ]).map((mat: any) => (
              <div key={mat.material} className="bg-slate-50 dark:bg-slate-800/40 rounded-xl p-3 border border-slate-100 dark:border-slate-700/50">
                <p className="font-medium text-[10px] font-bold text-slate-500 uppercase tracking-widest mb-2">{mat.material}</p>
                <p className="font-bold text-sm text-slate-900 dark:text-white tracking-tight">KSh {mat.avgPrice?.toFixed(2) || mat.avgPrice} <span className="font-bold text-[10px] text-slate-500">/kg</span></p>
                <span className={`text-[10px] ${mat.trend >= 0 ? 'text-emerald-500' : 'text-rose-500'}`}>
                  {mat.trend >= 0 ? '↑' : '↓'} {Math.abs(mat.trend).toFixed(1)}%
                </span>
              </div>
            ))}
          </div>

          <div className="bg-emerald-50 dark:bg-emerald-500/5 border border-emerald-200 dark:border-emerald-900/30 rounded-xl p-4">
            <div className="flex items-center gap-2 mb-2">
              <Bot className="font-medium w-4 h-4 text-emerald-600 dark:text-emerald-400" />
              <p className="font-medium text-[10px] text-emerald-600 dark:text-emerald-400 uppercase tracking-widest">AI Recommendation</p>
            </div>
            <p className="font-medium text-xs text-slate-700 dark:text-slate-300 leading-relaxed">
              Focus on collecting <strong>{highestDemand?.material_name || highestDemand?.material || 'PET Bottles'}</strong> this week.
              High demand detected in Industrial Area with expected profit of KSh {highestDemand?.price_per_kg || highestDemand?.avgPrice || 42}/kg.
            </p>
            <button onClick={() => navigate('/admin/market-intelligence')} className="font-medium mt-2 text-[10px] text-emerald-600 dark:text-emerald-400 hover:underline flex items-center gap-1">
              View details <ArrowRight className="w-3 h-3" />
            </button>
          </div>
        </Card>

        {/* Attention Center */}
        <Card>
          <div className="flex items-center justify-between mb-4">
            <div className="flex items-center gap-2">
              <AlertTriangle className="font-medium w-4 h-4 text-slate-500" />
              <h3 className="text-[10px] font-bold text-slate-500 uppercase tracking-widest">Attention Center</h3>
            </div>
            <button className="font-medium text-[10px] text-primary hover:underline flex items-center gap-1">
              View all alerts <ArrowRight className="w-3 h-3" />
            </button>
          </div>

          <div className="space-y-3">
            {alerts.length > 0 ? alerts.map((alert, idx) => (
              <div key={idx} className="flex items-center gap-3 p-3 bg-slate-50 dark:bg-slate-800/40 rounded-xl border border-slate-100 dark:border-slate-700/50">
                <div className={`w-8 h-8 rounded-lg flex items-center justify-center bg-white dark:bg-slate-800/60 border border-slate-200 dark:border-slate-700/50 shrink-0`}>
                  <alert.icon className={`w-4 h-4 ${alert.color}`} />
                </div>
                <div className="flex-1 min-w-0">
                  <p className="font-medium text-sm text-slate-900 dark:text-white truncate">{alert.title}</p>
                  <p className="font-bold text-[10px] text-slate-500 dark:text-slate-500">{alert.desc}</p>
                </div>
                <button
                  onClick={() => {
                    if (alert.actionLabel === 'Deposit') {
                      setShowDepositModal(true);
                    } else if (alert.action) {
                      navigate(alert.action);
                    }
                  }}
                  className={`text-[10px] ${alert.color} hover:underline flex items-center gap-1 shrink-0`}
                >
                  {alert.actionLabel} <ArrowRight className="w-3 h-3" />
                </button>
              </div>
            )) : (
              <div className="flex flex-col items-center justify-center py-10 text-center">
                <CheckCircle2 className="font-medium w-8 h-8 text-emerald-400 mb-2" />
                <p className="font-medium text-sm text-slate-900 dark:text-white">All Clear!</p>
                <p className="font-bold text-[10px] text-slate-500 mt-1">No pending alerts at this time</p>
              </div>
            )}
          </div>
        </Card>
      </div>

      {/* ── DEPOSIT MODAL ── */}
      <AnimatePresence>
        {showDepositModal && (
          <div className="fixed inset-0 z-[200] flex items-center justify-center p-4">
            <motion.div
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
              exit={{ opacity: 0 }}
              onClick={() => setShowDepositModal(false)}
              className="absolute inset-0 bg-slate-900/60 backdrop-blur-sm"
            />
            <motion.div
              initial={{ opacity: 0, scale: 0.9, y: 20 }}
              animate={{ opacity: 1, scale: 1, y: 0 }}
              exit={{ opacity: 0, scale: 0.9, y: 20 }}
              className="relative w-full max-w-sm bg-white dark:bg-slate-800 rounded-3xl shadow-2xl overflow-hidden p-8 border border-slate-100 dark:border-slate-800"
            >
              <div className="flex items-center justify-between mb-6">
                <h3 className="text-sm font-bold text-slate-900 dark:text-white">Wallet Deposit</h3>
                <button
                  onClick={() => setShowDepositModal(false)}
                  className="w-9 h-9 rounded-full bg-slate-100 dark:bg-slate-800 flex items-center justify-center"
                >
                  <X className="font-medium w-4 h-4 text-slate-500" />
                </button>
              </div>

              <div className="space-y-5">
                <div className="space-y-2">
                  <label className="font-medium text-[10px] capitalize tracking-[0.2em] text-slate-500 px-1">Enter Amount (KSh)</label>
                  <div className="relative">
                    <div className="font-medium absolute left-4 top-1/2 -translate-y-1/2 text-slate-500 text-sm">KSh</div>
                    <input
                      type="number"
                      autoFocus
                      value={depositAmount}
                      onChange={(e) => setDepositAmount(e.target.value)}
                      placeholder="0.00"
                      className="font-medium w-full h-14 pl-14 pr-5 bg-slate-50 dark:bg-slate-800 rounded-xl border border-slate-200 dark:border-slate-700 focus:ring-2 focus:ring-amber-500 focus:border-transparent text-xl text-slate-900 dark:text-white"
                    />
                  </div>
                </div>

                <button
                  disabled={isDepositing || !depositAmount}
                  onClick={handleDeposit}
                  className="font-medium w-full h-14 bg-amber-500 text-white rounded-xl text-sm shadow-lg shadow-amber-500/20 active:scale-95 transition-all disabled:opacity-50 disabled:active:scale-100 flex items-center justify-center gap-2"
                >
                  {isDepositing ? <Loader2 className="w-5 h-5 animate-spin" /> : 'Confirm Deposit'}
                </button>

                <p className="font-medium text-[10px] text-center text-slate-500">
                  Secured by Klinflow Payment Gateway
                </p>
              </div>
            </motion.div>
          </div>
        )}
      </AnimatePresence>
    </div>
  );
}
